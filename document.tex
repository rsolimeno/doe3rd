\documentclass{article}%


\usepackage{xparse}
\ExplSyntaxOn
\DeclareExpandableDocumentCommand \makealph { m }
{
	\int_to_alph:n { #1 }
}
\ExplSyntaxOff

\renewcommand\thesection{\Alph{section}}
\renewcommand\thesubsection{\makealph{subsection}}


\usepackage{multido}
\newcommand{\forLoop}[4][1]{\multido{\i=#2+#1}{#3}{#4}}


\begin{document}
	
	\section{Introduction}
	\makealph{27}
	\section{Discussion}

	\section{Example}
	% Exceed the range 1 to 26, i.e. -1 to 30
	\forLoop[1]{1}{26}
	{%
		\subsection{In the loop}
	}
	

\end{document}
